// THE NEW FORMATTING ENGINE async function exportFormattedExcel(dataArray) { const workbook = new ExcelJS.Workbook(); const sheet = workbook.addWorksheet("Annexure B"); // 1. Write the raw array data into the sheet sheet.addRows(dataArray); // Calculate where the footers ended up const totalRows = sheet.rowCount; const footerStartRow = totalRows - 2; // 2. Apply Custom Merges for your specific schema sheet.mergeCells('A1:W1'); // Main Title sheet.mergeCells('A2:W2'); // Subtitle sheet.mergeCells('A3:E3'); // PR No & Date sheet.mergeCells('F3:W3'); // Value of Indent // Vertical merges for standalone headers (Row 5 & 6) const verticalCols = ['A','B','C','D','E','F','G','H','I','J','P','U','V','W']; verticalCols.forEach(col => sheet.mergeCells(`${col}5:${col}6`)); // Horizontal merges for grouped headers sheet.mergeCells('K5:O5'); // Actual Consumption in last 5 Years sheet.mergeCells('Q5:R5'); // Pipeline PO sheet.mergeCells('S5:T5'); // Pipeline PR // Merge cells for the new footers sheet.mergeCells(`A${footerStartRow}:W${footerStartRow}`); // Asterisk note sheet.mergeCells(`B${footerStartRow + 1}:W${footerStartRow + 1}`); // Note 1 sheet.mergeCells(`B${footerStartRow + 2}:W${footerStartRow + 2}`); // Note 2 // 3. Apply Borders, Fonts, and Alignment sheet.eachRow((row, rowNumber) => { row.eachCell((cell, colNumber) => { // Format Headers and Data Rows (Grid style) if (rowNumber < footerStartRow) { cell.border = { top: {style:'thin'}, left: {style:'thin'}, bottom: {style:'thin'}, right: {style:'thin'} }; cell.alignment = { vertical: 'middle', horizontal: 'center', wrapText: true }; // Bold the headers (Rows 1 to 6) if (rowNumber <= 6) { cell.font = { bold: true }; } } // Format Footer Rows (Clean style) else { cell.alignment = { vertical: 'middle', horizontal: 'left', wrapText: true }; // Bold the word "Note" which is in Column 1 (A) of the second footer row if (rowNumber === footerStartRow + 1 && colNumber === 1) { cell.font = { bold: true }; } } }); }); // 4. Adjust Column Widths so it looks clean upon opening sheet.columns.forEach(col => { col.width = 12; }); sheet.getColumn('D').width = 35; // Item Description needs more space sheet.getColumn('W').width = 35; // Justification needs more space // 5. Trigger the download const buffer = await workbook.xlsx.writeBuffer(); saveAs(new Blob([buffer]), "Annexure_B_Formatted.xlsx"); }
