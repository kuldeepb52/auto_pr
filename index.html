Script Main HTML file - single page app for the procurement mapping tool index.html Core mapping logic JavaScript file - handles all data processing and Excel generation javascript // ============================================================ // SAP Procurement Annexure Generator — mapping.js // ------------------------------------------------------------ // WHAT THIS FILE DOES (for learning): // // 1. Reads uploaded Excel files using the SheetJS (XLSX) library. // 2. Parses both SAP reports into in-memory JavaScript objects. // 3. Applies "mapping logic" — the same rules you used to do by // hand — to produce one row per material. // 4. Builds two brand-new Excel workbooks (Annexure A & B) with // proper headers, formatting, and merged cells. // 5. Triggers browser downloads for both files. // ============================================================ // ── State ────────────────────────────────────────────────── const state = { me2m: null, zmm: null }; // ============================================================ // FILE HANDLING // Drag-and-drop and file selection both end up here. // ============================================================ function handleFile(key, input) { const file = input.files[0]; if (!file) return; readExcel(key, file); } function handleDrop(e, key) { e.preventDefault(); document.getElementById(`zone-${key}`).classList.remove('dragover'); const file = e.dataTransfer.files[0]; if (file) readExcel(key, file); } function handleDrag(e, key) { e.preventDefault(); document.getElementById(`zone-${key}`).classList.add('dragover'); } function handleDragLeave(key) { document.getElementById(`zone-${key}`).classList.remove('dragover'); } // readExcel: uses FileReader to read bytes, then SheetJS to parse. function readExcel(key, file) { const reader = new FileReader(); reader.onload = (e) => { try { // XLSX.read converts raw bytes into a "workbook" object. const wb = XLSX.read(e.target.result, { type: 'array', cellDates: true }); // sheet_to_json converts the first sheet into an array of row objects. const ws = wb.Sheets[wb.SheetNames[0]]; const rows = XLSX.utils.sheet_to_json(ws, { defval: null }); state[key] = rows; // UI feedback const zone = document.getElementById(`zone-${key === 'me2m' ? 'me2m' : 'zmm'}`); zone.classList.add('loaded'); document.getElementById(`name-${key === 'me2m' ? 'me2m' : 'zmm'}`).textContent = `✓ ${file.name} (${rows.length} rows)`; checkReady(); } catch (err) { alert(`Error reading ${file.name}: ${err.message}`); } }; reader.readAsArrayBuffer(file); } function checkReady() { document.getElementById('btn-gen').disabled = !(state.me2m && state.zmm); } // ============================================================ // LOGGING HELPER // ============================================================ function log(msg, type = 'ok') { const el = document.getElementById('log'); el.classList.add('visible'); el.innerHTML += `
${msg}
`; el.scrollTop = el.scrollHeight; } function clearLog() { const el = document.getElementById('log'); el.innerHTML = ''; el.classList.add('visible'); } // ============================================================ // MAIN GENERATE FUNCTION // Called when user clicks "Generate Annexure A & B" // ============================================================ function generate() { clearLog(); document.getElementById('downloads').classList.remove('visible'); document.getElementById('dl-divider').style.display = 'none'; document.getElementById('preview-card').style.display = 'none'; log('▶ Starting data processing...', 'info'); try { // ── STEP 1: Parse ME2M ───────────────────────────────── log(' [1/6] Parsing ME2M (PO history)...', 'info'); const me2m = parseME2M(state.me2m); log(` ✓ ME2M: ${Object.keys(me2m.lastPO).length} unique materials, ${me2m.pipelinePO.size} with open pipeline PO`); // ── STEP 2: Parse ZMMMATHIST ─────────────────────────── log(' [2/6] Parsing ZMMMATHIST (stock & consumption)...', 'info'); const zmm = parseZMM(state.zmm); log(` ✓ ZMM: ${Object.keys(zmm.stock).length} materials with stock, ${Object.keys(zmm.pr).length} with pipeline PR`); // ── STEP 4: Build Annexure A ─────────────────────────── log(' [4/6] Building Annexure A (Estimation Sheet)...', 'info'); const wbA = buildAnnexureA(rows); log(' ✓ Annexure A built'); // ── STEP 5: Build Annexure B ─────────────────────────── log(' [5/6] Building Annexure B (Procurement Justification)...', 'info'); const wbB = buildAnnexureB(rows); log(' ✓ Annexure B built'); // ── STEP 6: Trigger downloads ────────────────────────── log(' [6/6] Preparing downloads...', 'info'); const dlA = workbookToDataURL(wbA); const dlB = workbookToDataURL(wbB); document.getElementById('dl-a').href = dlA; document.getElementById('dl-b').href = dlB; document.getElementById('downloads').classList.add('visible'); document.getElementById('dl-divider').style.display = 'block'; Click the download buttons above.`, 'ok'); } catch (err) { log(`❌ ERROR: ${err.message}`, 'error'); console.error(err); } } // ============================================================ // PARSE ME2M // Returns: // lastPO : { material -> { poNo, poItem, poDate, netPrice, vendor, orderUnit } } // pipelinePO: Set of materials with open/pending PO deliveries // ============================================================ function parseME2M(rows) { // Key insight: ME2M has MULTIPLE rows per material (one per PO line). // We need the MOST RECENT PO per material (highest Document Date). const byMaterial = {}; for (const r of rows) { const mat = clean(r['Material']); if (!mat) continue; const date = parseDate(r['Document Date']); if (!date) continue; if (!byMaterial[mat] || date > byMaterial[mat].date) { byMaterial[mat] = { date, poNo : fmtPO(r['Purchasing Document']), poItem : fmtItem(r['Item']), netPrice: toNum(r['Net price']), vendor : clean(r['Name of Vendor']), orderUnit: clean(r['Order Unit']), shortText: clean(r['Short Text']), }; } } // Pipeline PO: rows where "Still to be delivered (qty)" > 0 const pipelinePO = {}; for (const r of rows) { const mat = clean(r['Material']); if (!mat) continue; const stillDeliver = toNum(r['Still to be delivered (qty)']); if (stillDeliver > 0) { const poNo = fmtPO(r['Purchasing Document']); const poItem = fmtItem(r['Item']); const qty = stillDeliver; if (!pipelinePO[mat]) pipelinePO[mat] = []; pipelinePO[mat].push({ poNo, poItem, qty }); } } return { lastPO: byMaterial, pipelinePO }; } // ============================================================ // PARSE ZMMMATHIST // Returns: // stock : { material -> { description, matType, stockQty, fy4, fy3, fy2, prevFY, currFY } } // pr : { material -> [{ prNoSlNo, date, qty }] } // ============================================================ function parseZMM(rows) { const stock = {}; const pr = {}; for (const r of rows) { const mat = clean(r['Material']); if (!mat) continue; const docType = clean(r['Doc. Type']); if (docType === 'Stock') { // Each material has exactly one "Stock" row — this holds current stock & consumption. stock[mat] = { description: clean(r['Material Description']), matType : clean(r['Mat. Type']), stockQty : toNum(r['Stock']), fy4 : toNum(r['Cons-Curr Fin. Yr - 4']), fy3 : toNum(r['Cons-Curr Fin. Yr -3']), fy2 : toNum(r['Cons-Curr Fin. Yr - 2']), prevFY : toNum(r['Cons-Prev Fin. Yr.']), currFY : toNum(r['Cons-Curr Fin. Year']), mrpType : clean(r['MRP Type']), }; } else if (docType === 'PurRqs') { // Pipeline PR: material has one or more "PurRqs" rows. // Doc.Detail format: "0100254655/00740 *" → PR No / PR Item const detail = clean(r['Doc. Detail']) || ''; const qty = toNum(r['Quantity']); const date = parseDate(r['Req. date']); if (!pr[mat]) pr[mat] = []; pr[mat].push({ detail, qty, date }); } } return { stock, pr }; } // ============================================================ // BUILD ROWS // Merges both datasets into one array of objects, // one object per unique material. // ============================================================ function buildRows(me2m, zmm) { // Union of all materials across both sources const allMaterials = new Set([ ...Object.keys(me2m.lastPO), ...Object.keys(zmm.stock), ]); const rows = []; let sno = 1; for (const mat of allMaterials) { const po = me2m.lastPO[mat] || {}; const st = zmm.stock[mat] || {}; const prs = zmm.pr[mat] || []; const pos = me2m.pipelinePO[mat] || []; // Description: prefer ZMM (more reliable), fall back to ME2M Short Text const description = st.description || po.shortText || ''; // 5-year total consumption const fiveYrCons = (st.fy4||0) + (st.fy3||0) + (st.fy2||0) + (st.prevFY||0) + (st.currFY||0); // Pipeline PR: combine all into one string for display const pipelinePRStr = prs .filter(p => p.qty > 0) .map(p => `${p.detail} (Qty:${p.qty})`) .join('; ') || ''; const pipelinePRDate = prs.length > 0 ? prs[0].date : null; const pipelinePRQty = prs.reduce((s, p) => s + (p.qty || 0), 0); // Pipeline PO const pipelinePOStr = pos .map(p => `${p.poNo}/${p.poItem} (Qty:${p.qty})`) .join('; ') || ''; const pipelinePOQty = pos.reduce((s, p) => s + (p.qty || 0), 0); rows.push({ sno, material : mat, description, stockQty : st.stockQty ?? 0, unit : po.orderUnit || 'NO', lastPONo : po.poNo || '', lastPOItem : po.poItem || '', lastPODate : po.date || null, lpp : po.netPrice || 0, // Last Purchase Price estimatedRate: po.netPrice || 0, // Default to LPP fiveYrCons, fy4 : st.fy4 || 0, fy3 : st.fy3 || 0, fy2 : st.fy2 || 0, prevFY : st.prevFY || 0, currFY : st.currFY || 0, pipelinePOStr, pipelinePOQty, pipelinePRStr, pipelinePRDate, pipelinePRQty, matType : st.matType || '', }); sno++; } // Sort by material code for consistency rows.sort((a, b) => a.material.localeCompare(b.material)); rows.forEach((r, i) => r.sno = i + 1); return rows; } // ============================================================ // BUILD ANNEXURE A — Estimation Sheet // Columns: S No | Material | Description | Population | Stock Qty // | Qty Requested | UOM | Last PO No | Last PO Date // | LPP | Budgetary Offer | Estimated Rate | Total Value // ============================================================ function buildAnnexureA(rows) { const wb = XLSX.utils.book_new(); aoa.push([`PR ECM No. : ${ecmNo || 'XXXXXXXXXXXXXXXXXXXXXXX'}`]); // Column headers (row 6 in Excel) aoa.push([ 'S No.', 'Material', 'Material Description', 'Population', 'Stock Qty', 'Qty requested', 'Unit of Measure', 'Last PO No.', 'Last PO date', 'LPP', 'Budgetary offer', 'Estimated Rate', 'Total Value', ]); // Data rows for (const r of rows) { // Total Value = Qty requested * Estimated Rate // Since we don't have "Qty requested" from the SAP reports (it comes from the PR), // we leave it blank for the user to fill in. } // Convert a workbook to a data URL for download function workbookToDataURL(wb) { const buf = XLSX.write(wb, { bookType: 'xlsx', type: 'array' }); const blob = new Blob([buf], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' }); return URL.createObjectURL(blob); } // Parse a value that might be a Date object, ISO string, or serial number function parseDate(val) { if (!val) return null; if (val instanceof Date) return isNaN(val.getTime()) ? null : val; const d = new Date(val); return isNaN(d.getTime()) ? null : d; } // Format a date as DD-MM-YYYY for display inside Excel cells function fmtExcelDate(d) { if (!d) return ''; const date = d instanceof Date ? d : new Date(d); if (isNaN(date.getTime())) return ''; const dd = String(date.getDate()).padStart(2,'0'); const mm = String(date.getMonth()+1).padStart(2,'0'); const yyyy = date.getFullYear(); return `${dd}-${mm}-${yyyy}`; } // Format a date for header string display function fmtDateStr(val) { if (!val) return 'XXXXXXX'; // val from is YYYY-MM-DD const parts = val.split('-'); if (parts.length === 3) return `${parts[2]}-${parts[1]}-${parts[0]}`; String(val).trim() : String(Math.round(n)); } // Format PO item — SAP exports as float like 10.0 function fmtItem(val) { if (!val) return ''; const n = Number(val); return isNaN(n) ? String(val).trim() : String(Math.round(n)); } // Convert any numeric-ish value to a JS number (strips whitespace, handles null) function toNum(val) { if (val === null || val === undefined || val === '') return 0; const n = parseFloat(String(val).replace(/,/g, '').trim()); return isNaN(n) ? 0 : n; } // Trim and return a clean string, or '' if null/undefined function clean(val) { if (val === null || val === undefined) return ''; return String(val).trim(); }
